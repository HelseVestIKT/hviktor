Standard radiobutton

Radiobutton fra primeng stylet etter Hviktor. Sett inn label for å få tilhørende label til radiobutton, og sett inn ariaLabel slik at radiobutton er tilgjengelig for skjermleser.

                        
                        <p-radioButton inputId="category" value="radiobutton" [(ngModel)]="category"></p-radioButton>
                        <label for="category">Standard radiobutton</label>

Disabled radiobutton

Sett inn [disabled] for å definere når en checkbox er deaktivert.

                        
                        <p-radioButton inputId="category" value="radiobutton" [(ngModel)]="category" [disabled]="true"></p-radioButton>
                        <label for="category">Standard radiobutton</label>

Radiobuttons i form

Bruk fieldset- og legend-tag for å definere en gruppe med tilhørende radiobuttons.

Se primeng sin dokumentasjon for mer informasjon om dynamisk bruk av radiobuttons og Reactive Forms

Velg kategori 
                
                    <form [formGroup]="formGroup" class="p-400">
                        <fieldset class="flex f-column p-400 gap-500">
                            <legend>Velg kategori</legend>
                            <div *ngFor="let category of categories">
                                <p-radioButton inputId="category" [value]="category" formControlName="selectedCategory"></p-radioButton>
                                <label for="category"> { { category.name } } </label>
                            </div>
                        </fieldset>
                    </form>
                
            
                
                    import { Component } from '@angular/core';
                    import { FormControl, FormGroup } from '@angular/forms';

                    @Component({
                        selector: 'app-radiobutton',
                        templateUrl: './radiobutton.component.html',
                        styleUrls: ['./radiobutton.component.scss']
                    })

                    export class RadiobuttonComponent {

                        formGroup!: FormGroup;

                        categories: any[] = [
                            { name: 'Accounting', key: 'A' },
                            { name: 'Marketing', key: 'M' },
                            { name: 'Production', key: 'P' },
                            { name: 'Research', key: 'R' }
                        ];

                        ngOnInit() {
                            this.formGroup = new FormGroup({
                                selectedCategory: new FormControl()
                            });
                        }
                    }
