Standard

Standard checkbox med HVIKTOR stil. Sett inn label i p-checkbox for å få tilhørende label til checkbox, og sett inn ariaLabel slik at checkboxen er tilgjengelig for skjermleser 

                
                        <p-checkbox label="Standard checkbox" value="value" ariaLabel="standard checkbox" ></p-checkbox>

Checkbox i formgroup

Bruk fieldset- og legend-tag for å definere en gruppe med tilhørende checkboxer.

Se primeng sin dokumentasjon for mer informasjon om dynamisk bruk av checkbox og Reactive Forms

Velg alternativer 
                
                    <form [formGroup]="formGroup" class="flex f-column p-400 gap-500">
                        <legend>Velg alternativer</legend>
                        <p-checkbox label="Alternativ 1" formControlName="checkbox" value="alternativ 1" ariaLabel="alternativ 1" ></p-checkbox>
                        <p-checkbox label="Alternativ 2" formControlName="checkbox" value="alternativ 2" ariaLabel="alternativ 2" ></p-checkbox>
                    </form>
                
            
                
                    import { Component } from '@angular/core';
                    import { FormControl, FormGroup } from '@angular/forms';

                    @Component({
                        selector: 'app-checkbox',
                        templateUrl: './checkbox.component.html',
                        styleUrls: ['./checkbox.component.scss']
                    })
                    export class CheckboxComponent {
                    
                        formGroup!: FormGroup;

                        ngOnInit() {
                                this.formGroup = new FormGroup({
                                checkbox: new FormControl<string | null>(null)
                            });
                        }
                    }

Disabled checkbox

Sett inn [disabled] for å definere når en checkbox er deaktivert. 

                        
                                <p-checkbox label="Disabled checkbox" [disabled]="true" ariaLabel="disabled checkbox" ></p-checkbox>
